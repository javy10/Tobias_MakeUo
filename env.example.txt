# Supabase Configuration
# Copia este contenido a un archivo llamado .env.local en la raíz del proyecto
# y completa con tus credenciales de Supabase

NEXT_PUBLIC_SUPABASE_URL=tu_supabase_url_aqui
NEXT_PUBLIC_SUPABASE_ANON_KEY=tu_supabase_anon_key_aqui

# Cómo obtener estas credenciales:
# 1. Ve a https://supabase.com/dashboard
# 2. Crea un nuevo proyecto o selecciona uno existente
# 3. Ve a Settings > API
# 4. Copia la URL del proyecto y la clave anónima (anon key)
# 5. Habilita las políticas de Storage en la consola de Supabase
